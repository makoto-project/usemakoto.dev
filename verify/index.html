Verifying a DBOM Document

How verification works

Three layers, always run in order. Failing layer 1 doesn't skip layer 2 — you get all errors at once.

Layer 1
Available

Structural

Fetch the schema from usemakoto.dev/schema/v0.1.json and validate the document against it. Catches wrong field types, unsupported enum values, and missing required fields.

→
Layer 2
Available

Provenance

Walk the lineage array and verify each step's input_hash matches the previous step's output_hash. The final hash must equal source.hash.value.

→
Layer 3
Coming soon

Cryptographic

Verify the signature.value against a public key resolved from signature.signer. The signing spec is not yet published — see the stub below.

The schema is always fetched at runtime from the canonical URL. There is no bundled copy.

Provenance rules

①
lineage[0].input_hash must be "n/a" — the first step has no predecessor.
②
Each step's input_hash must equal the previous step's output_hash.
③
The final step's output_hash must equal source.hash.value.

A document with one lineage step is valid as long as rule ① and ③ both hold.

BASH curl + jsonschema CLI

jq can't do real JSON Schema validation. Use the jsonschema CLI (ships with the Python package), which fetches and resolves schemas from a URL natively.

pip install jsonschema[format-nongpl]
verify-dbom.sh 
#!/usr/bin/env bash
# verify-dbom.sh
# Usage: ./verify-dbom.sh example.dbom.json

set -euo pipefail

SCHEMA_URL="https://usemakoto.dev/schema/v0.1.json"
DBOM_FILE="${1:?Usage: $0 <dbom-file>}"
ERRORS=0

fail() { echo " ✗ $*" >&2; ERRORS=$((ERRORS + 1)); }
pass() { echo " ✓ $*"; }

echo "Verifying: $DBOM_FILE"

# ── 1. Structural ─────────────────────────────────────────────────────────────
echo ""
echo "[1/2] Structural"

if jsonschema --instance "$DBOM_FILE" "$SCHEMA_URL" 2>/tmp/jsonschema-errors; then
  pass "conforms to schema"
else
  fail "schema validation failed:"
  cat /tmp/jsonschema-errors >&2
fi

# ── 2. Provenance chain ───────────────────────────────────────────────────────
echo ""
echo "[2/2] Provenance"

DOC=$(cat "$DBOM_FILE")

FIRST_INPUT=$(echo "$DOC" | jq -r '.lineage[0].input_hash')
if [ "$FIRST_INPUT" = "n/a" ]; then
  pass "step 1 input_hash is n/a"
else
  fail "step 1 input_hash should be \"n/a\", got \"$FIRST_INPUT\""
fi

echo "$DOC" | jq -r '
  .lineage as $steps |
  range(1; ($steps | length)) |
  . as $i |
  if $steps[$i].input_hash != $steps[$i-1].output_hash then
    "FAIL step \($steps[$i].step) input_hash does not match step \($steps[$i-1].step) output_hash"
  else
    "PASS step \($steps[$i].step) input_hash matches step \($steps[$i-1].step) output_hash"
  end
' | while IFS=' ' read -r result rest; do
  if [ "$result" = "PASS" ]; then pass "$rest"
  else fail "$rest"; fi
done

FINAL_OUT=$(echo "$DOC" | jq -r '.lineage[-1].output_hash')
SOURCE_HASH=$(echo "$DOC" | jq -r '.source.hash.value')
if [ "$FINAL_OUT" = "$SOURCE_HASH" ]; then
  pass "final step output_hash matches source.hash.value"
else
  fail "final step output_hash does not match source.hash.value"
fi

# ── Result ────────────────────────────────────────────────────────────────────
echo ""
if [ "$ERRORS" -eq 0 ]; then
  echo "✓ PASSED"; exit 0
else
  echo "✗ FAILED ($ERRORS error(s))"; exit 1
fi

PYTHON jsonschema + requests

Uses Draft202012Validator with a live schema fetch. Exits 0 on pass, 1 on any failure.

pip install jsonschema[format-nongpl] requests
verify_dbom.py 
#!/usr/bin/env python3
"""
verify_dbom.py
Usage: python verify_dbom.py example.dbom.json
Exits 0 on pass, 1 on any failure.
"""

import json
import sys
from pathlib import Path

import jsonschema
import requests

SCHEMA_URL = "https://usemakoto.dev/schema/v0.1.json"


def load_schema() -> dict:
    resp = requests.get(SCHEMA_URL, timeout=10)
    resp.raise_for_status()
    return resp.json()


def verify_structure(dbom: dict, schema: dict) -> list[str]:
    validator = jsonschema.Draft202012Validator(
        schema,
        format_checker=jsonschema.FormatChecker(),
    )
    return [
        f"[{' → '.join(str(p) for p in e.absolute_path) or 'root'}] {e.message}"
        for e in validator.iter_errors(dbom)
    ]


def verify_provenance(dbom: dict) -> list[str]:
    errors = []
    steps = dbom.get("lineage", [])

    if not steps:
        return ["lineage is empty"]

    if steps[0].get("input_hash") != "n/a":
        errors.append(
            f"step 1 input_hash should be \"n/a\", got \"{steps[0].get('input_hash')}\""
        )

    for i in range(1, len(steps)):
        prev_out = steps[i - 1].get("output_hash")
        curr_in  = steps[i].get("input_hash")
        if prev_out != curr_in:
            errors.append(
                f"step {steps[i]['step']} input_hash ({curr_in!r}) "
                f"does not match step {steps[i-1]['step']} output_hash ({prev_out!r})"
            )

    final_out   = steps[-1].get("output_hash")
    source_hash = dbom.get("source", {}).get("hash", {}).get("value")
    if final_out != source_hash:
        errors.append(
            f"final step output_hash ({final_out!r}) "
            f"does not match source.hash.value ({source_hash!r})"
        )

    return errors


def verify(path: str) -> bool:
    dbom   = json.loads(Path(path).read_text())
    schema = load_schema()
    all_errors = []

    print(f"\nVerifying: {path}")

    print("\n[1/2] Structural")
    errs = verify_structure(dbom, schema)
    if errs:
        all_errors.extend(errs)
        for e in errs:
            print(f" ✗ {e}", file=sys.stderr)
    else:
        print(" ✓ conforms to schema")

    print("\n[2/2] Provenance")
    errs = verify_provenance(dbom)
    if errs:
        all_errors.extend(errs)
        for e in errs:
            print(f" ✗ {e}", file=sys.stderr)
    else:
        print(" ✓ hash chain is consistent")

    print()
    if not all_errors:
        print("✓ PASSED")
        return True
    print(f"✗ FAILED ({len(all_errors)} error(s))", file=sys.stderr)
    return False


if __name__ == "__main__":
    path = sys.argv[1] if len(sys.argv) > 1 else "example.dbom.json"
    sys.exit(0 if verify(path) else 1)

JS Node.js · ajv

compileAsync + loadSchema fetches the schema (and any $refs) at validation time. No local copy needed.

npm install ajv ajv-formats
verify-dbom.mjs 
#!/usr/bin/env node
// verify-dbom.mjs
// Usage: node verify-dbom.mjs example.dbom.json

import Ajv2020   from "ajv/dist/2020.js";
import addFormats from "ajv-formats";
import { readFileSync } from "fs";

const SCHEMA_URL = "https://usemakoto.dev/schema/v0.1.json";

async function loadSchema(uri) {
  const res = await fetch(uri, { signal: AbortSignal.timeout(10_000) });
  if (!res.ok) throw new Error(`Failed to fetch schema: HTTP ${res.status}`);
  return res.json();
}

async function buildValidator() {
  const ajv = new Ajv2020({ allErrors: true, loadSchema });
  addFormats(ajv);
  const schema = await loadSchema(SCHEMA_URL);
  return ajv.compileAsync(schema);
}

function verifyProvenance(dbom) {
  const errors = [];
  const steps  = dbom.lineage ?? [];

  if (steps.length === 0) return ["lineage is empty"];

  if (steps[0].input_hash !== "n/a") {
    errors.push(`step 1 input_hash should be "n/a", got "${steps[0].input_hash}"`);
  }

  for (let i = 1; i < steps.length; i++) {
    const prevOut = steps[i - 1].output_hash;
    const currIn  = steps[i].input_hash;
    if (prevOut !== currIn) {
      errors.push(
        `step ${steps[i].step} input_hash ("${currIn}") ` +
        `does not match step ${steps[i - 1].step} output_hash ("${prevOut}")`
      );
    }
  }

  const finalOut   = steps.at(-1).output_hash;
  const sourceHash = dbom.source?.hash?.value;
  if (finalOut !== sourceHash) {
    errors.push(
      `final step output_hash ("${finalOut}") ` +
      `does not match source.hash.value ("${sourceHash}")`
    );
  }

  return errors;
}

async function verify(filePath) {
  const dbom     = JSON.parse(readFileSync(filePath, "utf8"));
  const validate = await buildValidator();
  const allErrors = [];

  console.log(`\nVerifying: ${filePath}`);

  console.log("\n[1/2] Structural");
  if (!validate(dbom)) {
    const errs = validate.errors.map(e => `[${e.instancePath || "root"}] ${e.message}`);
    allErrors.push(...errs);
    errs.forEach(e => console.error(` ✗ ${e}`));
  } else {
    console.log(" ✓ conforms to schema");
  }

  console.log("\n[2/2] Provenance");
  const provErrs = verifyProvenance(dbom);
  if (provErrs.length) {
    allErrors.push(...provErrs);
    provErrs.forEach(e => console.error(` ✗ ${e}`));
  } else {
    console.log(" ✓ hash chain is consistent");
  }

  console.log();
  if (allErrors.length === 0) { console.log("✓ PASSED"); return true; }
  console.error(`✗ FAILED (${allErrors.length} error(s))`);
  return false;
}

const filePath = process.argv[2] ?? "example.dbom.json";
process.exit((await verify(filePath)) ? 0 : 1);

GO gojsonschema

gojsonschema accepts a schema URI directly and fetches it over HTTP — pass the URL, not a file path.

go get github.com/xeipuuv/gojsonschema
verify_dbom.go 
// verify_dbom.go
// Usage: go run verify_dbom.go example.dbom.json
package main

import (
	"encoding/json"
	"fmt"
	"os"

	"github.com/xeipuuv/gojsonschema"
)

const schemaURL = "https://usemakoto.dev/schema/v0.1.json"

type dbomDoc struct {
	Source struct {
		Hash struct {
			Value string `json:"value"`
		} `json:"hash"`
	} `json:"source"`
	Lineage []struct {
		Step       int    `json:"step"`
		InputHash  string `json:"input_hash"`
		OutputHash string `json:"output_hash"`
	} `json:"lineage"`
}

func verifyStructure(path string) []string {
	result, err := gojsonschema.Validate(
		gojsonschema.NewReferenceLoader(schemaURL),
		gojsonschema.NewReferenceLoader("file://"+path),
	)
	if err != nil {
		return []string{fmt.Sprintf("validation error: %v", err)}
	}
	if result.Valid() {
		return nil
	}
	errs := make([]string, len(result.Errors()))
	for i, e := range result.Errors() {
		errs[i] = fmt.Sprintf("[%s] %s", e.Field(), e.Description())
	}
	return errs
}

func verifyProvenance(raw []byte) []string {
	var doc dbomDoc
	if err := json.Unmarshal(raw, &doc); err != nil {
		return []string{fmt.Sprintf("could not parse document: %v", err)}
	}

	var errors []string
	steps := doc.Lineage

	if len(steps) == 0 {
		return []string{"lineage is empty"}
	}

	if steps[0].InputHash != "n/a" {
		errors = append(errors, fmt.Sprintf(
			`step 1 input_hash should be "n/a", got %q`, steps[0].InputHash,
		))
	}

	for i := 1; i < len(steps); i++ {
		if steps[i].InputHash != steps[i-1].OutputHash {
			errors = append(errors, fmt.Sprintf(
				"step %d input_hash (%q) does not match step %d output_hash (%q)",
				steps[i].Step, steps[i].InputHash, steps[i-1].Step, steps[i-1].OutputHash,
			))
		}
	}

	if steps[len(steps)-1].OutputHash != doc.Source.Hash.Value {
		errors = append(errors, fmt.Sprintf(
			"final step output_hash (%q) does not match source.hash.value (%q)",
			steps[len(steps)-1].OutputHash, doc.Source.Hash.Value,
		))
	}

	return errors
}

func verify(path string) bool {
	raw, err := os.ReadFile(path)
	if err != nil {
		fmt.Fprintf(os.Stderr, "error reading file: %v\n", err)
		return false
	}

	fmt.Printf("\nVerifying: %s\n", path)
	var allErrors []string

	fmt.Println("\n[1/2] Structural")
	if errs := verifyStructure(path); len(errs) > 0 {
		allErrors = append(allErrors, errs...)
		for _, e := range errs {
			fmt.Fprintf(os.Stderr, " ✗ %s\n", e)
		}
	} else {
		fmt.Println(" ✓ conforms to schema")
	}

	fmt.Println("\n[2/2] Provenance")
	if errs := verifyProvenance(raw); len(errs) > 0 {
		allErrors = append(allErrors, errs...)
		for _, e := range errs {
			fmt.Fprintf(os.Stderr, " ✗ %s\n", e)
		}
	} else {
		fmt.Println(" ✓ hash chain is consistent")
	}

	fmt.Println()
	if len(allErrors) == 0 {
		fmt.Println("✓ PASSED")
		return true
	}
	fmt.Fprintf(os.Stderr, "✗ FAILED (%d error(s))\n", len(allErrors))
	return false
}

func main() {
	path := "example.dbom.json"
	if len(os.Args) > 1 {
		path = os.Args[1]
	}
	if !verify(path) {
		os.Exit(1)
	}
}

Cryptographic signature verification

⚠️ Stub — not yet implementable

The signature.value field exists in the schema, but the Makoto spec does not yet define:

  • What input is signed (which fields, in what canonical form)
  • How verifiers resolve signer (e.g. github:username) to a public key

Once the signing spec is published at usemakoto.dev, this section will be updated with working implementations. Until then, a document that passes structural and provenance checks is well-formed but not cryptographically attested.

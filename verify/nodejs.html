JS Node.js verifier

Uses AJV's compileAsync to fetch the schema at validation time. No bundled copy.

verify-dbom.mjs
#!/usr/bin/env node
// verify-dbom.mjs — Usage: node verify-dbom.mjs example.dbom.json

import Ajv2020    from "ajv/dist/2020.js";
import addFormats from "ajv-formats";
import { readFileSync } from "fs";

const SCHEMA_URL = "https://usemakoto.dev/schema/v0.1.json";

async function loadSchema(uri) {
  const res = await fetch(uri, { signal: AbortSignal.timeout(10_000) });
  if (!res.ok) throw new Error(`HTTP ${res.status}`);
  return res.json();
}

function verifyProvenance(dbom) {
  const errors = [], steps = dbom.lineage ?? [];
  if (!steps.length) return ["lineage is empty"];
  if (steps[0].input_hash !== "n/a")
    errors.push(`step 1 input_hash should be "n/a", got "${steps[0].input_hash}"`);
  for (let i = 1; i < steps.length; i++)
    if (steps[i - 1].output_hash !== steps[i].input_hash)
      errors.push(`step ${steps[i].step} input_hash does not match step ${steps[i-1].step} output_hash`);
  if (steps.at(-1).output_hash !== dbom.source?.hash?.value)
    errors.push("final step output_hash does not match source.hash.value");
  return errors;
}

async function verify(filePath) {
  const dbom = JSON.parse(readFileSync(filePath, "utf8"));
  const ajv  = new Ajv2020({ allErrors: true, loadSchema });
  addFormats(ajv);
  const validate  = await ajv.compileAsync(await loadSchema(SCHEMA_URL));
  const allErrors = [];

  console.log(`\nVerifying: ${filePath}`);

  console.log("\n[1/2] Structural");
  if (!validate(dbom)) {
    const errs = validate.errors.map(e => `[${e.instancePath || "root"}] ${e.message}`);
    allErrors.push(...errs); errs.forEach(e => console.error(` ✗ ${e}`));
  } else { console.log(" ✓ conforms to schema"); }

  console.log("\n[2/2] Provenance");
  const provErrs = verifyProvenance(dbom);
  if (provErrs.length) {
    allErrors.push(...provErrs); provErrs.forEach(e => console.error(` ✗ ${e}`));
  } else { console.log(" ✓ hash chain is consistent"); }

  console.log();
  if (!allErrors.length) { console.log("✓ PASSED"); return true; }
  console.error(`✗ FAILED (${allErrors.length} error(s))`); return false;
}

process.exit((await verify(process.argv[2] ?? "example.dbom.json")) ? 0 : 1);

GO Go verifier

Uses gojsonschema, which fetches the schema URI over HTTP directly — no local copy.

verify_dbom.go
// verify_dbom.go — Usage: go run verify_dbom.go example.dbom.json
package main

import (
	"encoding/json"
	"fmt"
	"os"

	"github.com/xeipuuv/gojsonschema"
)

const schemaURL = "https://usemakoto.dev/schema/v0.1.json"

type dbomDoc struct {
	Source  struct{ Hash struct{ Value string `json:"value"` } `json:"hash"` } `json:"source"`
	Lineage []struct {
		Step       int    `json:"step"`
		InputHash  string `json:"input_hash"`
		OutputHash string `json:"output_hash"`
	} `json:"lineage"`
}

func verifyStructure(path string) []string {
	result, err := gojsonschema.Validate(
		gojsonschema.NewReferenceLoader(schemaURL),
		gojsonschema.NewReferenceLoader("file://"+path),
	)
	if err != nil { return []string{err.Error()} }
	if result.Valid() { return nil }
	errs := make([]string, len(result.Errors()))
	for i, e := range result.Errors() { errs[i] = fmt.Sprintf("[%s] %s", e.Field(), e.Description()) }
	return errs
}

func verifyProvenance(raw []byte) []string {
	var doc dbomDoc
	if err := json.Unmarshal(raw, &doc); err != nil { return []string{err.Error()} }

	var errors []string
	steps := doc.Lineage
	if len(steps) == 0 { return []string{"lineage is empty"} }
	if steps[0].InputHash != "n/a" {
		errors = append(errors, fmt.Sprintf(`step 1 input_hash should be "n/a", got %q`, steps[0].InputHash))
	}
	for i := 1; i < len(steps); i++ {
		if steps[i].InputHash != steps[i-1].OutputHash {
			errors = append(errors, fmt.Sprintf(
				"step %d input_hash does not match step %d output_hash",
				steps[i].Step, steps[i-1].Step,
			))
		}
	}
	if steps[len(steps)-1].OutputHash != doc.Source.Hash.Value {
		errors = append(errors, "final step output_hash does not match source.hash.value")
	}
	return errors
}

func verify(path string) bool {
	raw, _ := os.ReadFile(path)
	fmt.Printf("\nVerifying: %s\n", path)
	var all []string

	fmt.Println("\n[1/2] Structural")
	if errs := verifyStructure(path); len(errs) > 0 {
		all = append(all, errs...)
		for _, e := range errs { fmt.Fprintf(os.Stderr, " ✗ %s\n", e) }
	} else { fmt.Println(" ✓ conforms to schema") }

	fmt.Println("\n[2/2] Provenance")
	if errs := verifyProvenance(raw); len(errs) > 0 {
		all = append(all, errs...)
		for _, e := range errs { fmt.Fprintf(os.Stderr, " ✗ %s\n", e) }
	} else { fmt.Println(" ✓ hash chain is consistent") }

	fmt.Println()
	if len(all) == 0 { fmt.Println("✓ PASSED"); return true }
	fmt.Fprintf(os.Stderr, "✗ FAILED (%d error(s))\n", len(all)); return false
}

func main() {
	path := "example.dbom.json"
	if len(os.Args) > 1 { path = os.Args[1] }
	if !verify(path) { os.Exit(1) }
}
